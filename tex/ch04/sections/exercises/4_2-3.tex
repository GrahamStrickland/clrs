% Exercise 4.2-3

We have the following algorithm:

\begin{algorithmic}[1]
  \TITLE{\textsc{Strassen-Multiply}$(A, B)$}
  \STATE $n = A.\mathit{rows}$
  \STATE let $C$ be a new $n \times n$ matrix
  \IF{$n == 1$}
  \STATE $C_{11} = A_{11} \cdot B_{11}$
  \RETURN $C$
  \ELSIF{$2 \nmid n$}
  \FOR{$i = 1$ \TO $n$}
  \STATE $c_{i1} = a_{i1} \cdot b_{1i}$
  \ENDFOR
  \FOR{$j = 1$ \TO $n$}
  \STATE $c_{1j} = a_{1j} \cdot b_{j1}$
  \ENDFOR
  \STATE partition $A$, $B$, and $C$ by excluding top row and left column of each
         into $A_{\textit{lower}}$, $B_{\textit{lower}}$, and $C_{\textit{lower}}$, respectively
  \STATE $C_{\textit{lower}} = \textsc{Strassen-Multiply}(A_{\textit{lower}}, B_{\textit{lower}})$
  \ELSE
  \STATE partition $A$, $B$, and $C$ as in equations (4.9)
  \STATE $S_1 = B_{12} - B_{22}$
  \STATE $S_2 = A_{11} + A_{12}$
  \STATE $S_3 = A_{21} + A_{22}$
  \STATE $S_4 = B_{21} - B_{11}$
  \STATE $S_5 = A_{11} + A_{22}$
  \STATE $S_6 = B_{11} + B_{22}$
  \STATE $S_7 = A_{12} - A_{22}$
  \STATE $S_8 = B_{21} + B_{22}$
  \STATE $S_9 = A_{11} - A_{21}$
  \STATE $S_{10} = B_{11} + B_{12}$
  \STATE $P_1 = \textsc{Strassen-Multiply}(A_{11}, S_1)$
  \STATE $P_2 = \textsc{Strassen-Multiply}(S_2, B_{22})$
  \STATE $P_3 = \textsc{Strassen-Multiply}(S_3, B_{11})$
  \STATE $P_4 = \textsc{Strassen-Multiply}(A_{22}, S_4)$
  \STATE $P_5 = \textsc{Strassen-Multiply}(S_5, S_6)$
  \STATE $P_6 = \textsc{Strassen-Multiply}(S_7, S_8)$
  \STATE $P_7 = \textsc{Strassen-Multiply}(S_9, S_{10})$
  \STATE $C_{11} = P_5 + P_4 - P_2 + P_6$
  \STATE $C_{12} = P_1 + P_2$
  \STATE $C_{21} = P_3 + P_4$
  \STATE $C_{22} = P_5 + P_1 - P_3 - P_7$
  \RETURN $C$
  \ENDIF
\end{algorithmic}

The second case $2 \nmid n$ adds a single step which takes $2n$ time, so that the 
algorithm still runs in time $\Theta(n^{\lg{7}} + 2n) = \Theta(n^{\lg{7}})$, 
since the running time of $\textsc{Strassen-Multiply}$ is $\Theta(n^{\lg{7}})$.
