% Exercise A.1-3

\begin{proof}
    We have
    \begin{equation*}
        \begin{split}
            \frac{d}{dx} \sum_{k = 0}^\infty k x^k &= \sum_{k = 1}^\infty k^2 x^{k - 1} \\
                                                   &= \frac{d}{dx} \frac{x}{{(1 - x)}^2} \\
                                                   &= \frac{{(1 - x)}^2 + 2x(1 - x)}{{(1 - x)}^4} \\
                                                   &= \frac{(1 - x) + 2x}{{(1 - x)}^3} \\
                                                   &= \frac{x + 1}{{(1 - x)}^3}.
        \end{split}
    \end{equation*}
    Thus we have
    \begin{equation*}
        \begin{split}
            x\sum_{k = 0}^\infty k^2 x^{k - 1} &= \sum_{k = 0}^\infty k^2 x^k \\
                                               &= \frac{x(x + 1)}{{(1 - x)}^3},
        \end{split}
    \end{equation*}
    for $\lvert x \rvert < 1$.
\end{proof}
