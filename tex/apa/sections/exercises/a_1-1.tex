\begin{equation*}
    \begin{split}
        \sum_{k = 1}^n (2k - 1) &= \sum_{k = 1}^n 2k - \sum_{k = 1}^n 1 \\
                                &= 2 \sum_{k = 1}^n k - \sum_{k = 1}^n 1 \\
                                &= n(n + 1) - n \\
                                &= n^2.
    \end{split}
\end{equation*}
