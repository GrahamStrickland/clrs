\begin{proof}
    For all $a, b, c > 0 \in \mathbb{R}$, we have
    \begin{equation*}
        \begin{split}
            \log_b{c} &= \frac{\log_a{c}}{\log_a{b}} \\
                      &= \log_b{a} \cdot \log_a{c} \\
                      &= \log_a{c} \cdot \log_b{a}.
        \end{split}
    \end{equation*}
    Thus we have
    \begin{equation*}
        \begin{split}
            a^{\log_b{c}} &= a^{\log_a{c} \cdot \log_b{a}} \\
                          &= {(a^{log_a{c}})}^{\log_b{a}} \\
                          &= c^{\log_b{a}}.
        \end{split}
    \end{equation*}
\end{proof}
